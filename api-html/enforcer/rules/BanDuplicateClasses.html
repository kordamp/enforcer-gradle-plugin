047 /**
048  * Bans duplicate classes on the classpath.
049  *
050  * Adapted from {@code org.apache.maven.plugins.enforcer.BanDuplicateClasses}.
051  * Original author: Robert Scholte
052  *
053  * @author Andres Almiray
054  * @since 0.1.0
055  */
056 @CompileStatic
057 class BanDuplicateClasses extends AbstractResolveDependencies {
058     /**
059      * Default ignores which are needed for JDK 9, cause in JDK 9 and above the <code>module-info.class</code> will be
060      * duplicated in any jar file. Furthermore in use cases for multi release jars the <code>module-info.class</code> is
061      * also contained several times.
062      */
063     private static final List<String> DEFAULT_CLASSES_IGNORES = ['module-info', 'META-INF/versions/*/module-info']
064 
065     final ListProperty<String> ignoreClasses
066     final ListProperty<String> configurations
067     final Property<Boolean> findAllDuplicates
068     final Property<Boolean> ignoreWhenIdentical
069 
070     final ListProperty<Dependency> dependencies
071 
072     @Inject
073     BanDuplicateClasses(ObjectFactory objects) {
074         super(objects, [AFTER_PROJECT, AFTER_PROJECTS] as EnforcerPhase[])
075         ignoreClasses = objects.listProperty(String).convention([])
076         dependencies = objects.listProperty(Dependency).convention([])
077         configurations = objects.listProperty(String).convention([])
078         findAllDuplicates = objects.property(Boolean).convention(false)
079         ignoreWhenIdentical = objects.property(Boolean).convention(false)
080     }
081 
082     @Override
083     protected void doExecute(EnforcerContext context) throws EnforcerRuleException {
084         if (context.enforcerPhase == AFTER_PROJECTS) {
085             enforceBanDuplicates(context)
086         } else {
087             enforceBanDuplicates(context)
088         }
089     }
090 
091     private void enforceBanDuplicates(EnforcerContext context) throws EnforcerRuleException {
092         Set<ResolvedArtifact> artifactsSeen = []
093         Map<String, ClassesWithSameName> all_classesSeen = [:]
094         Set<String> all_duplicateClassNames = []
095         context.project.configurations.each { Configuration c ->
096             Map<String, ClassesWithSameName> classesSeen = [:]
097             Set<String> duplicateClassNames = []
098             handleConfiguration(context, c, duplicateClassNames, classesSeen, artifactsSeen)
099             all_classesSeen.putAll(classesSeen)
100             all_duplicateClassNames.addAll(duplicateClassNames)
101         }
102         for (Project project : context.project.childProjects.values()) {
103             project.configurations.each { Configuration c ->
104                 Map<String, ClassesWithSameName> classesSeen = [:]
105                 Set<String> duplicateClassNames = []
106                 handleConfiguration(context, c, duplicateClassNames, classesSeen, artifactsSeen)
107                 all_classesSeen.putAll(classesSeen)
108                 all_duplicateClassNames.addAll(duplicateClassNames)
109             }
110         }
111         reportDuplicates(all_duplicateClassNames, all_classesSeen)
112     }

114     private void handleConfiguration(EnforcerContext context,
115                                      Configuration configuration,
116                                      Set<String> duplicateClassNames,
117                                      Map<String, ClassesWithSameName> classesSeen,
118                                      Set<ResolvedArtifact> artifactsSeen) {
119         if (!(configurations.get().empty) && !(configurations.get().contains(configuration.name))) {
120             return
121         }
122 
123         if (!configuration.canBeResolved) {
124             context.logger.debug("Configuration '${configuration.name}' cannot be resolved. Skipping duplicate class check.")
125             return
126         }
127 
128         context.logger.debug("Inspecting '${configuration.name}' for duplicate classes.")
129 
130         List<IgnorableDependency> ignorableDependencies = []
131 
132         IgnorableDependency ignoreableClasses = new IgnorableDependency()
133         ignoreableClasses.applyIgnoreClasses(DEFAULT_CLASSES_IGNORES)
134         if (ignoreClasses.get()) {
135             ignoreableClasses.applyIgnoreClasses((List<String>) ignoreClasses.get())
136         }
137         ignorableDependencies.add(ignoreableClasses)
138 
139         for (Dependency dependency : dependencies.get()) {
140             IgnorableDependency ignorableDependency = new IgnorableDependency()
141             if (isNotBlank(dependency.groupId)) {
142                 ignorableDependency.groupId = Pattern.compile(asRegex(dependency.groupId))
143             }
144             if (isNotBlank(dependency.artifactId)) {
145                 ignorableDependency.artifactId = Pattern.compile(asRegex(dependency.artifactId))
146             }
147             if (isNotBlank(dependency.version)) {
148                 ignorableDependency.version = Pattern.compile(asRegex(dependency.version))
149             }
150             if (isNotBlank(dependency.classifier)) {
151                 ignorableDependency.classifier = Pattern.compile(asRegex(dependency.classifier))
152             }
153             ignorableDependency.applyIgnoreClasses(dependency.ignoreClasses)
154             ignorableDependencies.add(ignorableDependency)
155         }
156 
157         configuration.resolve()
158 
159         Set<String> duplicates = []
160 
161         for (ResolvedArtifact o : configuration.resolvedConfiguration.resolvedArtifacts) {
162             if (artifactsSeen.contains(o)) {
163                 continue
164             }
165             artifactsSeen.add(o)
166 
167             File file = o.file
168             context.logger.debug('Searching for duplicate classes in ' + file)
169             if (file == null || !file.exists()) {
170                 context.logger.debug('Could not find ' + o + ' at ' + file)
171             } else if (file.directory) {
172                 try {
173                     for (String name : FileUtils.getFileNames(file, null, null, false)) {
174                         context.logger.debug('  ' + name)
175                         checkAndAddName(context, o, name, classesSeen, duplicates, ignorableDependencies)
176                     }
177                 } catch (IOException e) {
178                     throw fail('Unable to process dependency ' + o.toString() + ' due to ' + e.getLocalizedMessage(), e)
179                 }
180             } else if (isJarFile(o)) {
181                 try {
182                     JarFile jar = new JarFile(file)
183                     try {
184                         for (JarEntry entry : Collections.<JarEntry> list(jar.entries())) {
185                             checkAndAddName(context, o, entry.name, classesSeen, duplicates, ignorableDependencies)
186                         }
187                     } finally {
188                         try {
189                             jar.close()
190                         } catch (IOException e) {
191                             // ignore
192                         }
193                     }
194                 } catch (IOException e) {
195                     throw fail('Unable to process dependency ' + o.toString() + ' due to ' + e.getLocalizedMessage(), e)
196                 }
197             }
198         }
199 
200         duplicateClassNames.addAll(duplicates)
201     }

203     private void checkAndAddName(EnforcerContext context,
204                                  ResolvedArtifact artifact,
205                                  String pathToClassFile,
206                                  Map<String, ClassesWithSameName> classesSeen,
207                                  Set<String> duplicateClasses,
208                                  Collection<IgnorableDependency> ignores) throws EnforcerRuleException {
209         if (!pathToClassFile.endsWith('.class')) {
210             return
211         }
212 
213         for (IgnorableDependency c : ignores) {
214             if (c.matchesArtifact(artifact) && c.matches(pathToClassFile)) {
215                 if (classesSeen.containsKey(pathToClassFile)) {
216                     context.logger.debug('Ignoring excluded class ' + pathToClassFile)
217                 }
218                 return
219             }
220         }
221 
222         ClassesWithSameName classesWithSameName = classesSeen.get(pathToClassFile)
223         boolean isFirstTimeSeeingThisClass = (classesWithSameName == null)
224         ClassFile classFile = new ClassFile(pathToClassFile, artifact)
225 
226         if (isFirstTimeSeeingThisClass) {
227             classesSeen.put(pathToClassFile, new ClassesWithSameName(context, classFile))
228             return
229         }
230 
231         classesWithSameName.add(classFile)
232 
233         if (!classesWithSameName.hasDuplicates(ignoreWhenIdentical.get())) {
234             return
235         }
236 
237         if (findAllDuplicates.get()) {
238             duplicateClasses.add(pathToClassFile)
239         } else {
240             ResolvedArtifact previousArtifact = classesWithSameName.previous().getArtifactThisClassWasFoundIn()
241 
242             throw fail(new StringBuilder(!message.present ? 'Duplicate class found:' : message.get())
243                 .append(System.lineSeparator())
244                 .append(System.lineSeparator())
245                 .append('  Found in:')
246                 .append(System.lineSeparator())
247                 .append('    ')
248                 .append(previousArtifact)
249                 .append(System.lineSeparator())
250                 .append('    ')
251                 .append(artifact)
252                 .append(System.lineSeparator())
253                 .append('  Duplicate classes:')
254                 .append(System.lineSeparator())
255                 .append('    ')
256                 .append(pathToClassFile)
257                 .append(System.lineSeparator())
258                 .append("There may be others but 'findAllDuplicates' was set to false, so failing fast.")
259                 .append(System.lineSeparator())
260                 .append("Disable this rule temporarily with -D${resolveClassName()}.enabled=false and")
261                 .append(System.lineSeparator())
262                 .append("invoke 'dependencyInsight' or 'dependencies' to locate the source of the banned dependencies.")
263                 .toString())
264         }
265     }

267     private void reportDuplicates(Set<String> duplicateClassNames, Map<String, ClassesWithSameName> classesSeen) throws EnforcerRuleException {
268         if (!duplicateClassNames.isEmpty()) {
269             Map<Set<ResolvedArtifact>, List<String>> inverted = [:]
270             for (String className : duplicateClassNames) {
271                 ClassesWithSameName classesWithSameName = classesSeen.get(className)
272                 Set<ResolvedArtifact> artifactsOfDuplicateClass = classesWithSameName.getAllArtifactsThisClassWasFoundIn()
273 
274                 List<String> s = inverted.get(artifactsOfDuplicateClass)
275                 if (s == null) {
276                     s = []
277                 }
278                 s.add(classesWithSameName.toOutputString(ignoreWhenIdentical.get()))
279                 inverted.put(artifactsOfDuplicateClass, s)
280             }
281             StringBuilder buf = new StringBuilder(!message.present ? 'Duplicate classes found:' : message.get())
282             buf.append(System.lineSeparator())
283             for (Map.Entry<Set<ResolvedArtifact>, List<String>> entry : inverted.entrySet()) {
284                 buf.append(System.lineSeparator())
285                 buf.append('  Found in:');
286                 for (ResolvedArtifact a : entry.key) {
287                     buf.append(System.lineSeparator())
288                     buf.append('    ')
289                     buf.append(a)
290                 }
291                 buf.append(System.lineSeparator())
292                 buf.append('  Duplicate classes:')
293                 for (String classNameWithDuplicationInfo : entry.value) {
294                     buf.append(System.lineSeparator())
295                     buf.append('    ')
296                     buf.append(classNameWithDuplicationInfo)
297                 }
298                 buf.append(System.lineSeparator())
299             }
300             throw fail(buf.toString())
301         }
302     }

320     @Canonical
321     static class Dependency {
322         final String groupId
323         final String artifactId
324         final String version
325         final String classifier
326         final List<String> ignoreClasses = []
327 
328         void ignore(String className) {
329             if (isNotBlank(className)) ignoreClasses << className
330         }
331     }

333     private Dependency parseDependency(String str) {
334         if (isBlank(str)) {
335             throw illegalArgumentException('Unparseable dependency definition: empty input')
336         }
337 
338         String[] parts = str.split(':')
339         if (parts.length == 3) {
340             if (isBlank(parts[0])) {
341                 throw illegalArgumentException('Invalid dependency definition: empty groupId. ' + str)
342             }
343             if (isBlank(parts[1])) {
344                 throw illegalArgumentException('Invalid dependency definition: empty artifactId. ' + str)
345             }
346             if (isBlank(parts[2])) {
347                 throw illegalArgumentException('Invalid dependency definition: empty version. ' + str)
348             }
349             return new Dependency(parts[0].trim(), parts[1].trim(), parts[2].trim())
350         } else if (parts.length == 3) {
351             if (isBlank(parts[0])) {
352                 throw illegalArgumentException('Invalid dependency definition: empty groupId. ' + str)
353             }
354             if (isBlank(parts[1])) {
355                 throw illegalArgumentException('Invalid dependency definition: empty artifactId. ' + str)
356             }
357             if (isBlank(parts[2])) {
358                 throw illegalArgumentException('Invalid dependency definition: empty version. ' + str)
359             }
360             if (isBlank(parts[3])) {
361                 throw illegalArgumentException('Invalid dependency definition: empty classifier. ' + str)
362             }
363             return new Dependency(parts[0].trim(), parts[1].trim(), parts[2].trim(), parts[3].trim())
364         } else {
365             throw illegalArgumentException('Invalid dependency definition. ' + str)
366         }
367     }
368 
369     private Dependency parseDependency(Map<String, String> map) {
370         if (map.isEmpty()) {
371             throw illegalArgumentException('Unparseable dependency definition: empty input')
372         }
373 
374         String groupId = map.groupId
375         String artifactId = map.artifactId
376         String version = map.version
377         String classifier = map.classifier
378 
379         if (isBlank(groupId)) {
380             throw illegalArgumentException('Invalid dependency definition: empty groupId. ' + map)
381         }
382         if (isBlank(artifactId)) {
383             throw illegalArgumentException('Invalid dependency definition: empty artifactId. ' + map)
384         }
385         if (isBlank(version)) {
386             throw illegalArgumentException('Invalid dependency definition: empty version. ' + map)
387         }
388         if (isBlank(classifier) && map.containsKey('classifier')) {
389             throw illegalArgumentException('Invalid dependency definition: empty classifier. ' + map)
390         }
391 
392         return new Dependency(groupId.trim(), artifactId.trim(), version.trim(), classifier?.trim())
393     }
