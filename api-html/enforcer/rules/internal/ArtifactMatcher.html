020 import groovy.transform.CompileStatic
021 import org.apache.maven.artifact.versioning.ArtifactVersion
022 import org.apache.maven.artifact.versioning.DefaultArtifactVersion
023 import org.apache.maven.artifact.versioning.InvalidVersionSpecificationException
024 import org.apache.maven.artifact.versioning.Restriction
025 import org.apache.maven.artifact.versioning.VersionRange
026 import org.gradle.api.artifacts.ResolvedArtifact
027 
028 import static java.util.Objects.requireNonNull
029 import static org.apache.commons.lang3.StringUtils.isBlank
030 import static org.apache.maven.artifact.versioning.VersionRange.createFromVersionSpec
031 
032 /**
033  * This class is used for matching Artifacts against a list of patterns.
034  *
035  * Adapted from {@code org.apache.maven.plugins.enforcer.ArtifactMatcher}.
036  * Original author:  Jakub Senko
037  *
038  * @author Andres Almiray
039  * @since 0.1.0
040  */
041 @CompileStatic
042 final class ArtifactMatcher {
043     /**
044      * @author I don't know
045      */
046     static class Pattern {
047         private String pattern
048         private String[] parts
049 
050         Pattern(String pattern) {
051             this.pattern = requireNonNull(pattern, 'pattern')
052 
053             parts = pattern.split(":", 5)
054 
055             if (parts.length == 5) {
056                 throw new IllegalArgumentException('Pattern contains too many delimiters.')
057             }
058 
059             for (String part : parts) {
060                 if (isBlank(part)) {
061                     throw new IllegalArgumentException('Pattern or one of its part is empty.')
062                 }
063             }
064         }
065 
066         boolean match(ResolvedArtifact artifact) throws InvalidVersionSpecificationException {
067             requireNonNull(artifact, 'artifact')
068 
069             switch (parts.length) {
070                 case 4:
071                     if (!matches(parts[3], artifact.classifier)) {
072                         return false
073                     }
074                 case 3:
075                     if (!matches(parts[2], artifact.moduleVersion.id.version)) {
076                         if (!containsVersion(createFromVersionSpec(parts[2]),
077                             new DefaultArtifactVersion(artifact.moduleVersion.id.version))) {
078                             return false
079                         }
080                     }
081                 case 2:
082                     if (!matches(parts[1], artifact.moduleVersion.id.name)) {
083                         return false
084                     }
085                 case 1:
086                     return matches(parts[0], artifact.moduleVersion.id.group)
087                 default:
088                     throw new AssertionError()
089             }
090         }
091 
092         private boolean matches(String expression, String input) {
093             String regex = expression.replace('.', '\\.')
094                 .replace('*', '.*')
095                 .replace(':', '\\:')
096                 .replace('?', '.')
097                 .replace('[', '\\[')
098                 .replace(']', '\\]')
099                 .replace('(', '\\(')
100                 .replace(')', '\\)')
101 
102             // TODO: Check if this can be done better or prevented earlier.
103             if (input == null) {
104                 input = ''
105             }
106 
107             return java.util.regex.Pattern.matches(regex, input)
108         }
109 
110         /**
111          * Copied from Artifact.VersionRange. This is tweaked to handle singular ranges properly. Currently the default
112          * containsVersion method assumes a singular version means allow everything. This method assumes that "2.0.4" ==
113          * "[2.0.4,)"
114          *
115          * @param allowedRange range of allowed versions.
116          * @param theVersion the version to be checked.
117          * @return true if the version is contained by the range.
118          */
119         boolean containsVersion(VersionRange allowedRange, ArtifactVersion theVersion) {
120             boolean matched = false
121             ArtifactVersion recommendedVersion = allowedRange.getRecommendedVersion()
122             if (recommendedVersion == null) {
123                 List<Restriction> restrictions = allowedRange.getRestrictions()
124                 for (Restriction restriction : restrictions) {
125                     if (restriction.containsVersion(theVersion)) {
126                         matched = true
127                         break
128                     }
129                 }
130             } else {
131                 // only singular versions ever have a recommendedVersion
132                 @SuppressWarnings('unchecked')
133                 int compareTo = recommendedVersion.compareTo(theVersion)
134                 matched = (compareTo <= 0)
135             }
136             return matched
137         }
138 
139         @Override
140         String toString() {
141             return pattern
142         }
143     }
144 
145     private Collection<Pattern> patterns = []
146 
147     private Collection<Pattern> ignorePatterns = []
148 
149     /**
150      * Construct class by providing patterns as strings. Empty strings are ignored.
151      *
152      * @throws NullPointerException if any of the arguments is null
153      */
154     ArtifactMatcher(final Collection<String> patterns, final Collection<String> ignorePatterns) {
155         requireNonNull(patterns, 'patterns')
156         requireNonNull(ignorePatterns, 'ignorePatterns')
157 
158         for (String pattern : patterns) {
159             if (pattern != null && !''.equals(pattern)) {
160                 this.patterns.add(new Pattern(pattern))
161             }
162         }
163 
164         for (String ignorePattern : ignorePatterns) {
165             if (ignorePattern != null && !''.equals(ignorePattern)) {
166                 this.ignorePatterns.add(new Pattern(ignorePattern))
167             }
168         }
169     }
170 
171     /**
172      * Check if artifact matches patterns.
173      *
174      * @throws InvalidVersionSpecificationException
175      */
176     boolean match(ResolvedArtifact artifact)
177         throws InvalidVersionSpecificationException {
178         for (Pattern pattern : patterns) {
179             if (pattern.match(artifact)) {
180                 for (Pattern ignorePattern : ignorePatterns) {
181                     if (ignorePattern.match(artifact)) {
182                         return false
183                     }
184                 }
185                 return true
186             }
187         }
188         return false
189     }
190 }
