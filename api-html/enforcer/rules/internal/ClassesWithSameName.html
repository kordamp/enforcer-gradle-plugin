018 package enforcer.rules.internal
019 
020 import groovy.transform.CompileStatic
021 import org.gradle.api.artifacts.ResolvedArtifact
022 import org.kordamp.gradle.plugin.enforcer.api.EnforcerContext
023 
024 /**
025  * Represents one or more class files that have the same exact name.
026  *
027  * In this case the class name is a relative, file system path to the
028  * class file. For example:  org/apache/maven/Stuff.class
029  *
030  * Example of how we can have two of the same class:
031  *   - mockito-core-1.9.5.jar contains org/mockito/Mockito.class
032  *   - mockito-all-1.9.5.jar contains org/mockito/Mockito.class
033  *
034  * With that example you're not supposed to have both on the classpath. Typically
035  * you'd choose the maven way (mockito-core) or the convenient-for-non-maven-users
036  * way (mockito-all) but not both.
037  *
038  * Adapted from {@code org.apache.maven.plugins.enforcer.ClassesWithSameName}.
039  *
040  * @author Andres Almiray
041  * @since 0.1.0
042  */
043 @CompileStatic
044 class ClassesWithSameName {
045     private final EnforcerContext context
046     /** the path to the .class file. Example: org/apache/maven/Stuff.class */
047     private final String classFilePath
048     private final List<ClassFile> list = []
049 
050     /**
051      * @param context (required) the enforcer context
052      * @param initialClassFile (required) we require at least one class file. Splitting this param from the
053      *                         next one lets us require at least one at compile time (instead of runtime).
054      * @param additionalClassFiles (optional) additional class files
055      */
056     ClassesWithSameName(EnforcerContext context, ClassFile initialClassFile, ClassFile... additionalClassFiles) {
057         this.context = context
058         classFilePath = initialClassFile.getClassFilePath()
059         list.add(initialClassFile)
060 
061         for (ClassFile classFile : additionalClassFiles) {
062             throwIfClassNameDoesNotMatch(classFile, classFilePath)
063             list.add(classFile)
064         }
065     }
066 
067     /**
068      * @return the previous ClassFile, meaning, the one added before the most recent one. Psuedo-code:
069      *         add('Class1.class')
070      *         add('Class2.class')
071      *         previous()   // returns 'Class1.class'
072      */
073     ClassFile previous() {
074         if (list.size() > 1) {
075             int lastIndex = list.size() - 2
076             return list.get(lastIndex)
077         } else {
078             throw new IllegalArgumentException('there was only ' + list.size()
079                 + ' element(s) in the list, so there is no 2nd-to-last element to retrieve ')
080         }
081     }
082 
083     /**
084      * Add a new .class file with the same exact path and name as the other classes this file represents
085      * (though the artifact can be different).
086      * @param classFile The path to the .class file. Example: org/apache/maven/Stuff.class
087      */
088     void add(ClassFile classFile) {
089         throwIfClassNameDoesNotMatch(classFile, classFilePath)
090         list.add(classFile)
091     }
092 
093     /**
094      * @return Return a Set rather than a List so we can use this as the key in another Map.
095      *         List.of(3,2,1) doesn't equal List.of(1,2,3) but Set.of(3,2,1) equals Set.of(1,2,3)
096      */
097     Set<ResolvedArtifact> getAllArtifactsThisClassWasFoundIn() {
098         Set<ResolvedArtifact> result = []
099 
100         for (ClassFile classFile : list) {
101             result.add(classFile.getArtifactThisClassWasFoundIn())
102         }
103 
104         return result
105     }
106 
107     /**
108      * Main logic to determine if this object represents more than one of the exact same class
109      * on the classpath.
110      * @param ignoreWhenIdentical True if we should ignore two or more classes when they have the
111      *                            exact same bytecode false means fail whenever there's more than
112      *                            one of the same class, regardless of bytecode.
113      * @return true if there are duplicates, false if not.
114      */
115     boolean hasDuplicates(boolean ignoreWhenIdentical) {
116         boolean compareJustClassNames = !ignoreWhenIdentical
117         if (compareJustClassNames) {
118             return list.size() > 1
119         }
120 
121         if (list.size() <= 1) {
122             return false
123         }
124 
125         String previousHash = list.get(0).getHash()
126         for (int i = 1; i < list.size(); i++) {
127             String currentHash = list.get(i).getHash()
128             if (!previousHash.equals(currentHash)) {
129                 return true
130             }
131         }
132 
133         context.logger.debug('ignoring duplicates of class ' + classFilePath + ' since the bytecode matches exactly')
134 
135         return false
136     }
137 
138     /**
139      * @param ignoreWhenIdentical True if we should ignore two or more classes when they have the
140      *                            exact same bytecode false means fail whenever there's more than
141      *                            one of the same class, regardless of bytecode.
142      * @return the output string displayed on the command line when there are duplicate classes.
143      *
144      *         Example (ignoreWhenIdentical = false):
145      *         org/apache/maven/Stuff.class
146      *
147      *         Example (ignoreWhenIdentical = true):
148      *         org/apache/maven/Stuff.class  -- the bytecode exactly matches in these: a.jar and b.jar
149      */
150     String toOutputString(boolean ignoreWhenIdentical) {
151         String result = classFilePath
152 
153         if (list.size() >= 2 && ignoreWhenIdentical) {
154             StringBuilder duplicationInfo = new StringBuilder()
155             for (Set<ResolvedArtifact> groupedArtifacts : groupArtifactsWhoseClassesAreExactMatch().values()) {
156                 if (groupedArtifacts.size() <= 1) {
157                     continue
158                 }
159 
160                 if (duplicationInfo.length() == 0) {
161                     duplicationInfo.append('  -- the bytecode exactly matches in these: ')
162                 } else {
163                     duplicationInfo.append(' and more exact matches in these: ')
164                 }
165 
166                 duplicationInfo.append(joinWithSeparator(groupedArtifacts, ' and '))
167             }
168 
169             result += duplicationInfo.toString()
170         }
171 
172         return result
173     }
174 
175     private static void throwIfClassNameDoesNotMatch(ClassFile classFile, String otherClassFilePath) {
176         if (!classFile.getClassFilePath().equals(otherClassFilePath)) {
177             throw new IllegalArgumentException('Expected class ' + otherClassFilePath
178                 + ' but got ' + classFile.getClassFilePath())
179         }
180     }
181 
182     private String joinWithSeparator(Set<ResolvedArtifact> artifacts, String separator) {
183         StringBuilder result = new StringBuilder()
184         boolean first = true
185         for (ResolvedArtifact artifact : artifacts) {
186             if (first) {
187                 first = false
188             } else {
189                 result.append(separator)
190             }
191 
192             result.append(artifact)
193         }
194 
195         return result.toString()
196     }
197 
198     private Map<String, Set<ResolvedArtifact>> groupArtifactsWhoseClassesAreExactMatch() {
199         Map<String, Set<ResolvedArtifact>> groupedArtifacts = new LinkedHashMap<>()
200 
201         for (ClassFile classFile : list) {
202             Set<ResolvedArtifact> artifacts = groupedArtifacts.get(classFile.getHash())
203             if (artifacts == null) {
204                 artifacts = new LinkedHashSet<>()
205             }
206             artifacts.add(classFile.getArtifactThisClassWasFoundIn())
207 
208             groupedArtifacts.put(classFile.getHash(), artifacts)
209         }
210 
211         return groupedArtifacts
212     }
