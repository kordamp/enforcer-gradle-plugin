18 package enforcer.rules.internal
19 
20 import groovy.transform.CompileStatic
21 import org.gradle.api.artifacts.ResolvedArtifact
22 
23 /**
24  * This class represents a binary class file.
25  *
26  * The path to the class file should be a relative, file system path to the
27  * actual file. Examples:
28  *
29  *   - CORRECT: org/apache/maven/Stuff.class
30  *   - NO:  /org/apache/maven/Stuff.class
31  *   - NO:  org.apache.maven.Stuff
32  *   - NO:  maven.jar!org.apache.maven.Stuff
33  *   - NO:  maven.jar!/org/apache/maven/Stuff.class
34  *   - NO:  /path/to/some/directory/org.apache.maven.Stuff
35  *   - NO:  /path/to/some/directory/org/apache/maven/Stuff.class
36  *
37  * The file must exist in either a directory or a jar file, but the path
38  * of the directory/jar is not included in the class file path. Rather,
39  * it's included in the Artifact.
40  *
41  * Adapted from {@code org.apache.maven.plugins.enforcer.ClassFile}.
42  *
43  * @author Andres Almiray
44  * @since 0.1.0
45  */
46 @CompileStatic
47 class ClassFile {
48     /** the path to the .class file. Example: org/apache/maven/Stuff.class */
49     private final String classFilePath
50     private final ResolvedArtifact artifactThisClassWasFoundIn
51     private final Hasher hasher
52     private String lazilyComputedHash
53 
54     /**
55      * Constructor.
56      * @param classFilePath path to the class file. Example: org/apache/maven/Stuff.class
57      * @param artifactThisClassWasFoundIn the maven artifact the class appeared in (example: a jar file)
58      */
59     ClassFile(String classFilePath, ResolvedArtifact artifactThisClassWasFoundIn) {
60         this.classFilePath = classFilePath
61         this.artifactThisClassWasFoundIn = artifactThisClassWasFoundIn
62         this.hasher = new Hasher(classFilePath)
63     }
64 
65     /**
66      * @return the path to the .class file. Example: org/apache/maven/Stuff.class
67      */
68     String getClassFilePath() {
69         return classFilePath
70     }
71 
72     /**
73      * @return the maven artifact the class appeared in (example: a jar file)
74      */
75     ResolvedArtifact getArtifactThisClassWasFoundIn() {
76         return artifactThisClassWasFoundIn
77     }
78 
79     /**
80      * @return a hash or checksum of the binary file. If two files have the same hash
81      * then they are the same binary file.
82      */
83     String getHash() {
84         if (lazilyComputedHash == null) {
85             lazilyComputedHash = hasher.generateHash(artifactThisClassWasFoundIn)
86         }
87 
88         return lazilyComputedHash
89     }
90 
91 }
