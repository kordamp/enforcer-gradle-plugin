033 /**
034  * Contains the common code to compare a version against a version range.
035  *
036  * Examples of valid version values
037  * <ul>
038  * <li><code>2.0.4</code> Version 2.0.4 and higher (different from Maven meaning)</li>
039  * <li><code>[2.0,2.1)</code> Versions 2.0 (included) to 2.1 (not included)</li>
040  * <li><code>[2.0,2.1]</code> Versions 2.0 to 2.1 (both included)</li>
041  * <li><code>[2.0.5,)</code> Versions 2.0.5 and higher</li>
042  * <li><code>(,2.0.5],[2.1.1,)</code> Versions up to 2.0.5 (included) and 2.1.1 or higher</li>
043  * </ul>
044  *
045  * Adapted from {@code org.apache.maven.plugins.enforcer.AbstractVersionEnforcer}.
046  * Original author: <a href="mailto:brianf@apache.org">Brian Fox</a>
047  *
048  * @author Andres Almiray
049  * @since 0.1.0
050  */
051 @CompileStatic
052 abstract class AbstractVersionEnforcerRule extends AbstractStandardEnforcerRule {
053     final Property<String> version
054 
055     AbstractVersionEnforcerRule(ObjectFactory objects) {
056         this(objects, EnforcerPhase.values())
057     }
058 
059     AbstractVersionEnforcerRule(ObjectFactory objects, EnforcerPhase[] allowedPhases) {
060         super(objects, allowedPhases)
061         version = objects.property(String)
062     }
063 
064     @Override
065     protected void doExecute(EnforcerContext context) throws EnforcerRuleException {
066         ArtifactVersion detectedVersion = detectVersion(context)
067         String requiredVersionRange = version.get()
068 
069         if (isBlank(requiredVersionRange)) {
070             throw new EnforcerRuleException(this.class, variableName + " version can't be empty.")
071         } else {
072             VersionRange vr = null
073             String msg = 'Detected ' + variableName + ' Version: ' + detectedVersion
074 
075             // short circuit check if the strings are exactly equal
076             if (detectedVersion.toString().equals(requiredVersionRange)) {
077                 context.logger.info(msg + ' is allowed in the range ' + requiredVersionRange + '.')
078             } else {
079                 try {
080                     vr = VersionRange.createFromVersionSpec(requiredVersionRange)
081 
082                     if (containsVersion(vr, detectedVersion)) {
083                         context.logger.info(msg + ' is allowed in the range ' + requiredVersionRange + '.')
084                     } else {
085                         String m = message.orNull
086 
087                         if (isBlank(m)) {
088                             m = msg + ' is not in the allowed range ' + vr + '.'
089                         }
090 
091                         throw fail(m)
092                     }
093                 } catch (InvalidVersionSpecificationException e) {
094                     throw fail('The requested ' + variableName + ' version '
095                         + requiredVersionRange + ' is invalid.', e)
096                 }
097             }
098         }
099     }
100 
101     protected abstract ArtifactVersion detectVersion(EnforcerContext context)
102 
103     protected abstract String getVariableName()
104 
105     /**
106      * Copied from Artifact.VersionRange. This is tweaked to handle singular ranges properly. Currently the default
107      * containsVersion method assumes a singular version means allow everything. This method assumes that "2.0.4" ==
108      * "[2.0.4,)"
109      *
110      * @param allowedRange range of allowed versions.
111      * @param theVersion the version to be checked.
112      * @return true if the version is contained by the range.
113      */
114     static boolean containsVersion(VersionRange allowedRange, ArtifactVersion theVersion) {
115         boolean matched = false
116         ArtifactVersion recommendedVersion = allowedRange.getRecommendedVersion()
117         if (recommendedVersion == null) {
118             List<Restriction> restrictions = allowedRange.getRestrictions()
119             for (Restriction restriction : restrictions) {
120                 if (restriction.containsVersion(theVersion)) {
121                     matched = true
122                     break
123                 }
124             }
125         } else {
126             // only singular versions ever have a recommendedVersion
127             @SuppressWarnings('unchecked')
128             int compareTo = recommendedVersion.compareTo(theVersion)
129             matched = (compareTo <= 0)
130         }
131         return matched
132     }
133 }
