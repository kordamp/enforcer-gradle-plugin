18 package enforcer.rules
19 
20 import groovy.transform.CompileStatic
21 import org.gradle.api.model.ObjectFactory
22 import org.gradle.api.provider.Property
23 import org.kordamp.gradle.plugin.enforcer.api.EnforcerContext
24 
25 import javax.inject.Inject
26 
27 /**
28  * Rule to validate the main artifact is within certain size constraints.
29  *
30  * Adapted from {@code org.apache.maven.plugins.enforcer.RequireFilesSize}.
31  * Original authors: <a href="mailto:brianf@apache.org">Brian Fox</a, Roman Stumm
32  *
33  * @author Andres Almiray
34  * @since 0.1.0
35  */
36 @CompileStatic
37 class RequireFilesSize extends AbstractRequireFiles {
38     final Property<Long> maxsize
39     final Property<Long> minsize
40     private String errorMsg
41 
42     @Inject
43     RequireFilesSize(ObjectFactory objects) {
44         super(objects)
45         maxsize = objects.property(Long).convention(10_000L)
46         minsize = objects.property(Long).convention(0L)
47     }
48 
49     @Override
50     protected boolean checkFile(EnforcerContext context, File file) {
51         if (file == null) {
52             // if we get here and it's null, treat it as a success.
53             return true
54         }
55 
56         // check the file now
57         if (file.exists()) {
58             long length = file.length()
59             if (length < minsize.get()) {
60                 this.errorMsg = (file.name + ' size (' + length + ') too small. Min. is ' + minsize.get())
61                 return false
62             } else if (length > maxsize.get()) {
63                 this.errorMsg = (file.name + ' size (' + length + ') too large. Max. is ' + maxsize.get())
64                 return false
65             } else {
66                 context.logger.info(file.name
67                     + ' size ('
68                     + length
69                     + ') is OK ('
70                     + (minsize.get() == maxsize.get() || minsize.get() == 0 ? ('max. ' + maxsize.get())
71                     : ('between ' + minsize.get() + ' and ' + maxsize.get())) + ' byte).')
72 
73                 return true
74             }
75         } else {
76             this.errorMsg = (file.absolutePath + ' does not exist!')
77             return false
78         }
79     }
80 
81     @Override
82     protected String getErrorMsg() {
83         return this.errorMsg
84     }
85 }
