021 import org.apache.commons.codec.digest.DigestUtils
022 import org.codehaus.plexus.util.IOUtil
023 import org.gradle.api.model.ObjectFactory
024 import org.gradle.api.provider.ListProperty
025 import org.gradle.api.provider.Property
026 import org.kordamp.gradle.plugin.enforcer.api.AbstractEnforcerRule
027 import org.kordamp.gradle.plugin.enforcer.api.EnforcerContext
028 import org.kordamp.gradle.plugin.enforcer.api.EnforcerPhase
029 import org.kordamp.gradle.plugin.enforcer.api.EnforcerRuleException
030 
031 import javax.inject.Inject
032 
033 import static org.apache.commons.lang3.StringUtils.isBlank
034 import static org.kordamp.gradle.plugin.enforcer.api.EnforcerPhase.BEFORE_BUILD
035 
036 /**
037  * Rule to validate a file to match the specified checksum.
038  *
039  * Adapted from {@code org.apache.maven.plugins.enforcer.RequireFileChecksum}.
040  * Original authors: Edward Samson, Lyubomyr Shaydariv
041  *
042  * @author Andres Almiray
043  * @since 0.1.0
044  */
045 @CompileStatic
046 class RequireFileChecksum extends AbstractEnforcerRule {
047     final Property<String> message
048     final ListProperty<EnforcerPhase> phases
049     final Property<File> file
050     final Property<String> checksum
051     final Property<String> type
052 
053     @Inject
054     RequireFileChecksum(ObjectFactory objects) {
055         super(objects)
056         message = objects.property(String)
057         phases = objects.listProperty(EnforcerPhase).convention([BEFORE_BUILD])
058         checksum = objects.property(String)
059         type = objects.property(String)
060         file = objects.property(File)
061     }
062 
063     @Override
064     void execute(EnforcerContext context) throws EnforcerRuleException {
065         if (context.enforcerPhase in phases.get()) {
066             context.logger.debug("Enforcing rule ${resolveClassName()} on ${context}")
067             doExecute(context)
068         }
069     }
070 
071     protected void doExecute(EnforcerContext context) throws EnforcerRuleException {
072         if (!file.present) {
073             throw fail('Input file unspecified')
074         }
075 
076         if (!type.present) {
077             throw fail('Hash type unspecified. Valid value may be one of [md5, sha1, sha256, sha384, sha512]')
078         }
079 
080         if (!checksum.present) {
081             throw fail('Checksum unspecified')
082         }
083 
084         if (!file.get().exists()) {
085             String message = message.orNull
086             if (isBlank(message)) {
087                 message = 'File does not exist: ' + file.get().getAbsolutePath()
088             }
089             throw fail(message)
090         }
091 
092         if (file.get().isDirectory()) {
093             throw fail('Cannot calculate the checksum of directory: '
094                 + file.get().getAbsolutePath())
095         }
096 
097         if (!file.get().canRead()) {
098             throw fail('Cannot read file: ' + file.get().getAbsolutePath())
099         }
100 
101         String checksum = calculateChecksum()
102         println "checksum ${this.checksum.get()}"
103         println "checksum ${checksum}"
104 
105         if (!checksum.equalsIgnoreCase(this.checksum.get())) {
106             String exceptionMessage = message.orNull
107             if (isBlank(exceptionMessage)) {
108                 exceptionMessage = type.get() + ' hash of ' + file.get() + ' was ' + checksum +
109                     ' but expected ' + this.checksum.get()
110             }
111             throw fail(exceptionMessage)
112         }
113     }
114 
115     private String calculateChecksum() throws EnforcerRuleException {
116         InputStream inputStream = null
117         try {
118             inputStream = new FileInputStream(file.get())
119             if ('md5'.equals(type.get())) {
120                 return DigestUtils.md5Hex(inputStream)
121             } else if ('sha1'.equals(type.get())) {
122                 return DigestUtils.shaHex(inputStream)
123             } else if ('sha256'.equals(type.get())) {
124                 return DigestUtils.sha256Hex(inputStream)
125             } else if ('sha384'.equals(type.get())) {
126                 return DigestUtils.sha384Hex(inputStream)
127             } else if ('sha512'.equals(type.get())) {
128                 return DigestUtils.sha512Hex(inputStream)
129             } else {
130                 throw fail('Unsupported hash type: ' + type.get())
131             }
132         } catch (IOException e) {
133             throw fail('Unable to calculate checksum', e)
134         } finally {
135             IOUtil.close(inputStream)
136         }
137     }
138 }
