018 package enforcer.rules
019 
020 import groovy.transform.CompileStatic
021 import org.gradle.api.Project
022 import org.gradle.api.artifacts.Configuration
023 import org.gradle.api.artifacts.ResolvedArtifact
024 import org.gradle.api.model.ObjectFactory
025 import org.gradle.api.provider.ListProperty
026 import org.kordamp.gradle.plugin.enforcer.api.EnforcerContext
027 import org.kordamp.gradle.plugin.enforcer.api.EnforcerPhase
028 import org.kordamp.gradle.plugin.enforcer.api.EnforcerRuleException
029 
030 import static org.apache.commons.lang3.StringUtils.isNotBlank
031 
032 /**
033  * Abstract Rule for banning dependencies.
034  *
035  * Adapted from {@code org.apache.maven.plugins.enforcer.AbstractBanDependencies}.
036  * Original author: <a href="mailto:brianf@apache.org">Brian Fox</a>
037  *
038  * @author Andres Almiray
039  * @since 0.1.0
040  */
041 @CompileStatic
042 abstract class AbstractBanDependencies extends AbstractStandardEnforcerRule {
043     /**
044      * Optional list of dependency configurations to search.
045      */
046     final ListProperty<String> configurations
047 
048     AbstractBanDependencies(ObjectFactory objects) {
049         this(objects, EnforcerPhase.values())
050     }
051 
052     AbstractBanDependencies(ObjectFactory objects, EnforcerPhase[] allowedPhases) {
053         super(objects, allowedPhases)
054         configurations = objects.listProperty(String).convention([])
055     }
056 
057     @Override
058     protected void doExecute(EnforcerContext context) throws EnforcerRuleException {
059         // get the list of dependencies
060         Set<ResolvedArtifact> dependencies = getDependenciesToCheck(context)
061 
062         // look for banned dependencies
063         Set<ResolvedArtifact> foundExcludes = checkDependencies(context, dependencies)
064 
065         // if any are found, fail the check but list all of them
066         if (foundExcludes != null && !foundExcludes.isEmpty()) {
067             String message = message.orNull
068 
069             StringBuilder buf = new StringBuilder()
070             if (isNotBlank(message)) {
071                 buf.append(message).append(System.lineSeparator())
072             }
073             for (ResolvedArtifact artifact : foundExcludes) {
074                 buf.append(getErrorMessage(artifact))
075             }
076             message = buf.append("Disable this rule temporarily with -D${resolveClassName()}.enabled=false and")
077                 .append(System.lineSeparator())
078                 .append("invoke 'dependencies' to locate the source of the banned dependencies.").toString()
079 
080             throw fail(message)
081         }
082     }
083 
084     protected CharSequence getErrorMessage(ResolvedArtifact artifact) {
085         return 'Found Banned Dependency: ' + toString(artifact) + System.lineSeparator()
086     }
087 
088     protected String toString(ResolvedArtifact artifact) {
089         StringBuilder b = new StringBuilder(artifact.moduleVersion.id.group)
090             .append(':')
091             .append(artifact.moduleVersion.id.name)
092             .append(':')
093             .append(artifact.moduleVersion.id.version)
094         if (isNotBlank(artifact.classifier)) {
095             b.append(':')
096                 .append(artifact.classifier)
097         }
098         b.toString()
099     }
100 
101     protected Set<ResolvedArtifact> getDependenciesToCheck(EnforcerContext context) {
102         Set<ResolvedArtifact> artifacts = []
103 
104         context.logger.info("Configurations to be checked (if empty then all resolvable configurations will be checked): ${configurations.get()}")
105 
106         context.project.configurations.each { Configuration c ->
107             handleConfiguration(context, context.project, c, artifacts)
108         }
109         for (Project project : context.project.childProjects.values()) {
110             project.configurations.each { Configuration c ->
111                 handleConfiguration(context, project, c, artifacts)
112             }
113         }
114 
115         artifacts
116     }
117 
118     protected void handleConfiguration(EnforcerContext context, Project project, Configuration configuration, Set<ResolvedArtifact> artifacts) {
119         if (!(configurations.get().empty) && !(configurations.get().contains(configuration.name))) {
120             return
121         }
122 
123         String prefix = project.path == ':' ? ':' : project.path + ':'
124         if (!configuration.canBeResolved) {
125             context.logger.debug("Configuration '${prefix}${configuration.name}' cannot be resolved. Skipping check.")
126             return
127         }
128 
129         Configuration cfg = configuration.copyRecursive()
130         context.logger.info("Resolving configuration ${prefix}${cfg.name}.")
131         cfg.resolve()
132 
133         context.logger.info("Configuration ${prefix}${cfg.name} contains ${cfg.resolvedConfiguration.resolvedArtifacts.size()} artifacts.")
134         artifacts.addAll(cfg.resolvedConfiguration.resolvedArtifacts)
135     }
136 
137     protected abstract Set<ResolvedArtifact> checkDependencies(EnforcerContext context, Set<ResolvedArtifact> artifacts)
138 }
